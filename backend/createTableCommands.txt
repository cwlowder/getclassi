CREATE TABLE Users (NetId VARCHAR(200) NOT NULL PRIMARY KEY, Name VARCHAR(200));
CREATE TABLE Classes (CRN VARCHAR(200) NOT NULL PRIMARY KEY, Title VARCHAR(200), Department VARCHAR(200), Instructor VARCHAR(200));
CREATE TABLE Events(EventId int NOT NULL PRIMARY KEY, CRN VARCHAR(200) NOT NULL, DueDate VARCHAR(200) NOT NULL, Event_Des VARCHAR(2000) NOT NULL, FOREIGN KEY(CRN) REFERENCES Classes(CRN));
CREATE TABLE Enrollments(CRN VARCHAR(200) NOT NULL, NetId VARCHAR(200) NOT NULL, FOREIGN KEY(CRN) REFERENCES Classes(CRN), FOREIGN KEY(NetId) REFERENCES Users(NetId), PRIMARY KEY(CRN, NetId));
